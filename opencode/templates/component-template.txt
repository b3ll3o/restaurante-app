'use client';

import { type ReactNode } from 'react';
import { cn } from '@/lib/utils';

/**
 * Props do componente {NomeDoComponente}
 */
export interface {NomeDoComponente}Props {
  /** children do componente */
  children?: ReactNode;
  /** classe CSS adicional */
  className?: string;
  /** disabled state */
  disabled?: boolean;
  /** callback quando clicado */
  onClick?: () => void;
}

/**
 * Componente {NomeDoComponente}
 *
 * @example
 * ```tsx
 * <{NomeDoComponente} onClick={() => console.log('clicked')}>
 *   children
 * </{NomeDoComponente}>
 * ```
 */
export function {NomeDoComponente}({
  children,
  className,
  disabled = false,
  onClick,
}: {NomeDoComponente}Props) {
  return (
    <div
      className={cn(
        'relative inline-flex items-center justify-center',
        disabled && 'opacity-50 cursor-not-allowed',
        className
      )}
      onClick={disabled ? undefined : onClick}
      role="button"
      tabIndex={disabled ? -1 : 0}
    >
      {children}
    </div>
  );
}
